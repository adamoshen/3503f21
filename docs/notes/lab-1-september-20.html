1.2.8 Facetting 

weather %>%
  filter(Month %in% c("09", "10", "11")) %>%
  mutate(Precipitation = Total.Rain.mm. + 25 * Total.Snow.cm.) %>%
  ggplot(aes(x=Year, y=Precipitation))+
  geom_line(aes(colour=Month))+
  geom_smooth(method = "loess", formula = y ~ x)+
  facet_wrap(~Month)
